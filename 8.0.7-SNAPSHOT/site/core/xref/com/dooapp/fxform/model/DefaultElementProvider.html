34  /**
35   * Default implementation used to create elements for a bean.
36   * <p/>
37   * User: Antoine Mischler <antoine@dooapp.com>
38   * Date: 03/12/2013
39   * Time: 14:51
40   */
41  public class DefaultElementProvider implements ElementProvider {
42  
43      private final static Logger logger = Logger.getLogger(DefaultElementProvider.class.getName());
44  
45      FieldProvider fieldProvider = new ReflectionFieldProvider();
46  
47      ElementFactory elementFactory = new DefaultElementFactory();
48  
49      private final String[] includes;
50  
51      private List<FieldFilter> filters;
52  
53      public DefaultElementProvider(String... includes) {
54          this(new ReflectionFieldProvider(), includes);
55      }
56  
57      public DefaultElementProvider(FieldProvider fieldProvider, String... includes) {
58          this(new DefaultElementFactory(), fieldProvider, includes);
59      }
60  
61      public DefaultElementProvider(ElementFactory elementFactory, FieldProvider fieldProvider, String... includes) {
62          this.elementFactory = elementFactory;
63          this.fieldProvider = fieldProvider;
64          this.includes = includes;
65          this.filters = new LinkedList<FieldFilter>();
66          this.filters.add(new SyntheticFieldFilter());
67          this.filters.add(new NonVisualFieldFilter());
68      }
69  
70      @Override
71      public <T> ListProperty<Element> getElements(final ObjectProperty<T> source) {
72          final ListProperty<Element> elements = new SimpleListProperty<Element>(FXCollections.<Element>observableArrayList());
73          elements.setAll(createElements(source));
74          return elements;
75      }
76  
77      protected <T> ListProperty<Element> createElements(final ObjectProperty<T> source) {
78          ListProperty<Element> elements = new SimpleListProperty<Element>(FXCollections.<Element>observableArrayList());
79          List<Field> fields = applyFilters(extractFields(source.get()));
80          // Create elements for the resulting field list
81          for (Field field : fields) {
82              try {
83                  final Element element = elementFactory.create(field);
84                  element.sourceProperty().bind(new ObjectBinding() {
85  
86                      {
87                          bind(source);
88                      }
89  
90                      @Override
91                      protected Object computeValue() {
92                          if (source.get() != null && source.get() instanceof MultipleBeanSource) {
93                              MultipleBeanSource multipleBeanSource = (MultipleBeanSource) source.get();
94                              return multipleBeanSource.getSource(element);
95                          }
96                          return source.get();
97                      }
98                  });
99                  if (element.getType() != null) {
100                     elements.add(element);
101                 }
102             } catch (FormException e) {
103                 logger.log(Level.WARNING, e.getMessage(), e);
104             }
105         }
106         return elements;
107     }
108 
109     private List<Field> applyFilters(List<Field> fields) {
110         List<Field> filtered = new ArrayList<Field>();
111         for (Field field : fields) {
112             boolean accept = true;
113             for (FieldFilter filter : filters) {
114                 accept = accept && filter.accept(field);
115             }
116             if (accept) {
117                 filtered.add(field);
118             }
119         }
120         return filtered;
121     }
122 
123     private <T> List<Field> extractFields(T source) {
124         if (includes != null && includes.length > 0) {
125             return fieldProvider.getProperties(source, Arrays.asList(includes));
126         } else {
127             return fieldProvider.getProperties(source);
128         }
129     }
130 
131     public List<FieldFilter> getFilters() {
132         return filters;
133     }
