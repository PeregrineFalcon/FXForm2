34  /**
35   * Skin for the MapPropertyControl.
36   * <p/>
37   * Created at 14/10/13 17:39.<br>
38   *
39   * @author Antoine Mischler <antoine@dooapp.com>

41  public class MapPropertySkin implements Skin<MapPropertyControl> {
42      /**
43       * The logger
44       */
45      private static final Logger logger = LoggerFactory.getLogger(MapPropertySkin.class);
46  
47      private final MapPropertyControl control;
48  
49      GridPane gridPane = new GridPane();
50  
51      int index = 0;
52  
53      private Map<String, Integer> indexMap = new HashMap<String, Integer>();
54  
55      BorderPane root = new BorderPane();
56      private MapChangeListener<String, String> mapChangeListener = new MapChangeListener<String, String>() {
57          @Override
58          public void onChanged(Change<? extends String, ? extends String> change) {
59              if (change.wasAdded()) {
60                  addEntry(change.getKey(), change.getValueAdded());
61              }
62              if (change.wasRemoved()) {
63                  removeEntry(change.getKey(), change.getValueRemoved());
64              }
65          }
66      };
67  
68      public MapPropertySkin(MapPropertyControl control) {
69          this.control = control;
70          control.mapProperty().addListener(new ChangeListener<ObservableMap<String, String>>() {
71              @Override
72              public void changed(ObservableValue<? extends ObservableMap<String, String>> observableValue, ObservableMap<String, String> stringStringObservableMap, ObservableMap<String, String> stringStringObservableMap2) {
73                  if (stringStringObservableMap != null) {
74                      unconfigure(stringStringObservableMap);
75                  }
76                  if (stringStringObservableMap2 != null) {
77                      configure(stringStringObservableMap2);
78                  }
79              }
80          });
81          if (control.getMap() != null) {
82              configure(control.getMap());
83          }
84          final TextField keyTextField = new TextField();
85          final TextField valueTextField = new TextField();
86          gridPane.setHgap(10);
87          root.setCenter(gridPane);
88          root.setBottom(HBoxBuilder.create().children(keyTextField, valueTextField,
89                  ButtonBuilder.create().text("Add").onAction(new EventHandler<ActionEvent>() {
90                      @Override
91                      public void handle(ActionEvent actionEvent) {
92                          MapPropertySkin.this.control.getMap().put(keyTextField.getText(), valueTextField.getText());
93                      }
94                  }).build()).build());
95      }
96  
97      private void configure(ObservableMap<String, String> stringStringObservableMap2) {
98          for (String key : stringStringObservableMap2.keySet()) {
99              addEntry(key, stringStringObservableMap2.get(key));
100         }
101         stringStringObservableMap2.addListener(mapChangeListener);
102     }
103 
104     private void addEntry(final String key, final String value) {
105         Label keyLabel = new Label(key);
106         Label valueLabel = new Label(value);
107         Button button = new Button("-");
108         button.setOnAction(new EventHandler<ActionEvent>() {
109             @Override
110             public void handle(ActionEvent actionEvent) {
111                 removeEntry(key, value);
112             }
113         });
114         GridPane.setColumnIndex(keyLabel, 0);
115         GridPane.setColumnIndex(valueLabel, 1);
116         GridPane.setColumnIndex(button, 2);
117         GridPane.setRowIndex(keyLabel, index);
118         GridPane.setRowIndex(valueLabel, index);
119         GridPane.setRowIndex(button, index);
120         gridPane.getChildren().addAll(keyLabel, valueLabel, button);
121         indexMap.put(key, index);
122         index++;
123     }
124 
125     private void removeEntry(String key, String valueRemoved) {
126         removeIndex(indexMap.get(key));
127         indexMap.remove(key);
128     }
129 
130     private void removeIndex(int i) {
131         List<Node> toRemove = new LinkedList<Node>();
132         for (Node node : gridPane.getChildren()) {
133             if (GridPane.getRowIndex(node) == i) {
134                 toRemove.add(node);
135             }
136         }
137         for (Node node : toRemove) {
138             gridPane.getChildren().remove(node);
139         }
140     }
141 
142     private void unconfigure(ObservableMap<String, String> stringStringObservableMap) {
143         stringStringObservableMap.removeListener(mapChangeListener);
144         gridPane.getChildren().clear();
145         index = 0;
146     }
147 
148     @Override
149     public MapPropertyControl getSkinnable() {
150         return control;
151     }
152 
153     @Override
154     public Node getNode() {
155         return root;
156     }

158     @Override
159     public void dispose() {
160     }
