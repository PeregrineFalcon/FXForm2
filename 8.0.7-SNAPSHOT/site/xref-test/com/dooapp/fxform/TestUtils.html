33      /**
34       * Check whether a list of fields contains a field with the given name.
35       *
36       * @param name
37       * @param fields
38       * @return
39       */
40      public static boolean containsNamedField(String name, List<Element> fields) {
41          for (Element field : fields) {
42              if (name.equals(field.getName())) {
43                  return true;
44              }
45          }
46          return false;
47      }
48  
49      /**
50       * Get a list of test fields based from TestBean.
51       *
52       * @return
53       */
54      public static List<Element> getTestFields() {
55          List<Field> fields = new ReflectionFieldProvider().getProperties(new TestBean());
56          List<Element> elements = new LinkedList<Element>();
57          for (Field field: fields) {
58              try {
59                  elements.add(new ReadOnlyPropertyFieldElement(field));
60              } catch (FormException e) {
61                  e.printStackTrace();
62              }
63          }
64          return elements;
65      }
