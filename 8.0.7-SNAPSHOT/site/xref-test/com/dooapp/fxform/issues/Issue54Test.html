34  public class Issue54Test {
35  
36      public static class MyBean {
37  
38          public StringProperty property1 = new SimpleStringProperty();
39  
40          public StringProperty property2 = new SimpleStringProperty();
41  
42          public StringProperty property3 = new SimpleStringProperty();
43  
44      }
45  
46      private ResourceProvider tested;
47  
48      private Element element1;
49  
50      private Element element2;
51  
52      private Element element3;
53  
54      @Before
55      public void setup() throws NoSuchFieldException, FormException {
56          MyBean myBean = new MyBean();
57          tested = new DefaultResourceProvider();
58          tested.setResourceBundle(ResourceBundle.getBundle("Issue56Test"));
59          element1 = new PropertyFieldElement(MyBean.class.getDeclaredField("property1"));
60          element2 = new PropertyFieldElement(MyBean.class.getDeclaredField("property2"));
61          element3 = new PropertyFieldElement(MyBean.class.getDeclaredField("property3"));
62          element1.sourceProperty().set(myBean);
63          element2.sourceProperty().set(myBean);
64          element3.sourceProperty().set(myBean);
65      }
66  
67      @Test
68      public void testGetString() throws NoSuchFieldException, FormException {
69          // Test resource defined with bean name
70          Assert.assertEquals("Property1 label", tested.getString(element1, NodeType.LABEL).get());
71          Assert.assertEquals("Property1 tooltip", tested.getString(element1, NodeType.TOOLTIP).get());
72          // Test resource define without bean name
73          Assert.assertEquals("Property2 label", tested.getString(element2, NodeType.LABEL).get());
74          Assert.assertEquals("Property2 tooltip", tested.getString(element2, NodeType.TOOLTIP).get());
75          // Test default values
76          Assert.assertEquals("property3", tested.getString(element3, NodeType.LABEL).get());
77          Assert.assertEquals(null, tested.getString(element3, NodeType.TOOLTIP).get());
78      }
79  
80  
81      @Test
82      public void testUpdateResourceBundle() {
83          tested.setResourceBundle(ResourceBundle.getBundle("Issue56Test2"));
84          Assert.assertEquals("Property1 label2", tested.getString(element1, NodeType.LABEL).get());
85          Assert.assertEquals("Property1 tooltip2", tested.getString(element1, NodeType.TOOLTIP).get());
86          Assert.assertEquals("property2", tested.getString(element2, NodeType.LABEL).get());
87          Assert.assertEquals(null, tested.getString(element2, NodeType.TOOLTIP).get());
88          Assert.assertEquals("Property3 label2", tested.getString(element3, NodeType.LABEL).get());
89          Assert.assertEquals("Property3 tooltip2", tested.getString(element3, NodeType.TOOLTIP).get());
90      }
