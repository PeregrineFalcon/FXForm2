47  public class Demo extends Application {
48  
49      private FXForm<MyBean> fxForm;
50  
51      private StackPane root = new StackPane();
52  
53      private final String css = Demo.class.getResource("style.css").toExternalForm();
54  
55      protected void setup() {
56          MyBean joe = new MyBean("Joe", "contact@", "How does this crazy form works?", true, MyBean.Subject.QUESTION);
57          new ObjectPropertyObserver(joe);
58          // register a factory for the userMap field that is not handled by default
59          DefaultFactoryProvider.addGlobalFactory(new TypeFieldHandler(MapProperty.class), new MapPropertyFactory());
60          fxForm = new FXFormBuilder()
61                  .source(joe)
62                  .categorize("-USER-", "name", "welcome", "email", "-DATA-", "subject", "message")
63                  .resourceBundle(ResourceBundle.getBundle("com.dooapp.fxform.Demo"))
64                  .build();
65          fxForm.setTitle("Dude, where is my form?");
66          root.getChildren().add(createNode());
67      }
68  
69      private Node createNode() {
70          VBox vBox = new VBox();
71          vBox.getChildren().addAll(createSkinSelector(), createCSSNode(), fxForm, createConstraintNode());
72          return ScrollPaneBuilder.create().content(vBox).fitToWidth(true).build();
73      }
74  
75      private Node createConstraintNode() {
76          return ButtonBuilder.create().text("Validate")
77                  .defaultButton(true)
78                  .onAction(new EventHandler<ActionEvent>() {
79                      @Override
80                      public void handle(ActionEvent actionEvent) {
81                          ListView<ConstraintViolation> listView = new ListView<ConstraintViolation>();
82                          listView.setCellFactory(new Callback<ListView<ConstraintViolation>, ListCell<ConstraintViolation>>() {
83  
84                              @Override
85                              public ListCell<ConstraintViolation> call(ListView<ConstraintViolation> constraintViolation) {
86                                  return new ListCell<ConstraintViolation>() {
87                                      @Override
88                                      protected void updateItem(ConstraintViolation constraintViolation, boolean b) {
89                                          super.updateItem(constraintViolation, b);
90                                          if (constraintViolation != null) {
91                                              setText(constraintViolation.getPropertyPath().toString() + " - " + constraintViolation.getMessage());
92                                          } else {
93                                              setText("");
94                                          }
95                                      }
96                                  };
97                              }
98                          });
99                          listView.getItems().setAll(fxForm.getConstraintViolations());
100                         Scene scene = new Scene(listView);
101                         Stage stage = new Stage();
102                         stage.setScene(scene);
103                         stage.show();
104                     }
105                 }).build();
106     }
107 
108     private Node createCSSNode() {
109         CheckBox checkBox = new CheckBox("Use css");
110         checkBox.selectedProperty().addListener(new ChangeListener<Boolean>() {
111             public void changed(ObservableValue<? extends Boolean> observableValue, Boolean aBoolean, Boolean aBoolean1) {
112                 if (aBoolean1) {
113                     root.getScene().getStylesheets().add(css);
114                 } else {
115                     root.getScene().getStylesheets().remove(css);
116                 }
117             }
118         });
119         checkBox.setSelected(true);
120         return checkBox;
121     }
122 
123     /**
124      * Create a selector that changes the skin of the form.
125      *
126      * @return
127      */
128     private Node createSkinSelector() {
129         ChoiceBox<FXFormSkinFactory> choiceBox = new ChoiceBox<FXFormSkinFactory>();
130         choiceBox.getItems().addAll(FXFormSkinFactory.DEFAULT_FACTORY, FXFormSkinFactory.INLINE_FACTORY, new FXFormSkinFactory() {
131             @Override
132             public FXFormSkin createSkin(FXForm form) {
133                 return new FXMLSkin(form, Demo.class.getResource("Demo_form.fxml"));
134             }
135 
136             @Override
137             public String toString() {
138                 return "FXML Skin (Demo_form.fxml)";
139             }
140         });
141         choiceBox.getSelectionModel().selectedItemProperty().addListener(new ChangeListener<FXFormSkinFactory>() {
142             public void changed(ObservableValue<? extends FXFormSkinFactory> observableValue, FXFormSkinFactory fxFormSkinFactory, FXFormSkinFactory fxFormSkinFactory1) {
143                 fxForm.setSkin(fxFormSkinFactory1.createSkin(fxForm));
144             }
145         });
146         choiceBox.getSelectionModel().selectFirst();
147         return choiceBox;
148     }
