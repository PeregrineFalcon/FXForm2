37  /**
38   * A skin where elements are inlined (label, editor). The tooltip is displayed under the editor.
39   * User: Antoine Mischler <antoine@dooapp.com>
40   * Date: 24/08/11
41   * Time: 11:03
42   */
43  public class InlineSkin extends FXFormSkin {
44  
45      public InlineSkin(FXForm fxForm) {
46          super(fxForm);
47      }
48  
49      private Node createTitleNode() {
50          Label label = new AutoHidableLabel();
51          label.getStyleClass().add("form-title");
52          label.textProperty().bind(fxForm.titleProperty());
53          return label;
54      }
55  
56      protected GridPane gridPane;
57      protected int row = 0;
58      private Map<String, Node> categoryNode = new HashMap<String, Node>();
59  
60  
61  
62      @Override
63      protected Node createRootNode() throws NodeCreationException {
64          VBox titleBox = new VBox();
65          titleBox.getChildren().add(createTitleNode());
66          VBox contentBox = new VBox();
67          contentBox.setPadding(new Insets(5.0, 5.0, 5.0, 5.0));
68          contentBox.getStyleClass().add("form-content-box");
69          titleBox.getChildren().add(contentBox);
70          contentBox.setSpacing(5.0);
71          gridPane = GridPaneBuilder.create().hgap(5.0).vgap(5.0).build();
72          contentBox.getChildren().addAll(createClassLevelConstraintNode(), gridPane);
73          return titleBox;
74      }
75  
76      @Override
77      protected ElementNodes createElementNodes(Element element) {
78          FXFormNode editor = createEditor(element);
79          FXFormNode label = createLabel(element);
80          FXFormNode constraint = createConstraint(element);
81          FXFormNode tooltip = createTooltip(element);
82          GridPane.setHgrow(editor.getNode(), Priority.SOMETIMES);
83          gridPane.addRow(row++, label.getNode(), editor.getNode());
84          gridPane.add(constraint.getNode(), 1, row++);
85          gridPane.add(tooltip.getNode(), 1, row++);
86          return new ElementNodes(label, editor, tooltip, constraint);
87      }
88  
89      @Override
90      protected void deleteElementNodes(ElementNodes elementNodes) {
91          removeRow(GridPane.getRowIndex(elementNodes.getEditor().getNode()));
92          removeRow(GridPane.getRowIndex(elementNodes.getTooltip().getNode()));
93          removeRow(GridPane.getRowIndex(elementNodes.getConstraint().getNode()));
94      }
95  
96      @Override
97      protected void addCategory(String category) {
98          if (categoryMap.keySet().size() > 1) {
99              Node node = createCategoryNode(category);
100             categoryNode.put(category, node);
101             GridPane.setColumnSpan(node, 2);
102             gridPane.add(node, 0, row++);
103             row++;
104         }
105     }
106 
107     protected Node createCategoryNode(String category) {
108         return new Separator();
109     }
110 
111     @Override
112     protected void removeCategory(String category) {
113         if (categoryNode.containsKey(category)) {
114             removeRow(GridPane.getRowIndex(categoryNode.get(category)));
115             categoryNode.remove(category);
116         }
117     }
118 
119 
120     /**
121      * Remove a row by moving all nodes under this row one row up.
122      *
123      * @param row
124      */
125     protected void removeRow(int row) {
126         // copy children to another list since we are going to iterate on it and modify the children list
127         List<Node> children = new LinkedList<Node>(gridPane.getChildren());
128         for (Node node : children) {
129             int nodeRow = GridPane.getRowIndex(node);
130             if (nodeRow == row) {
131                 gridPane.getChildren().remove(node);
132             } else if (nodeRow > row) {
133                 GridPane.setRowIndex(node, nodeRow - 1);
134             }
135         }
136         this.row--;
137     }

139 
140     @Override
141     public String toString() {
142         return "Inline skin";
143     }
